#include <iostream>

template<typename T>
class Point {
public:
    T x, y;

    // Constructeur
    Point(T x = T{}, T y = T{}) : x(x), y(y) {}

    // Méthode membre ajoutée (zone de conflit possible)
    void translate(const T& dx, const T& dy) {
        x += dx;
        y += dy;
    }
};

// Opérateur + non-membre (signature conservée, implémentation modifiée)
template<typename T>
Point<T> operator+(const Point<T>& a, const Point<T>& b) {
    return Point<T>(a.x - b.x, a.y - b.y); // modification volontaire
}

// Opérateur << (structure intacte, format modifié)
template<typename T>
std::ostream& operator<<(std::ostream& os, const Point<T>& p) {
    return os << "[x=" << p.x << ", y=" << p.y << "]";
}

// Exemple d'utilisation
int main() {
    Point<int> p1(2, 3), p2(4, 5);
    Point<int> p3 = p1 + p2;
    std::cout << p3 << std::endl;

    Point<double> q1(1.5, 2.5), q2(0.5, 1.0);
    q1.translate(1.0, 1.0);
    std::cout << q1 + q2 << std::endl;

    return 0;
}
