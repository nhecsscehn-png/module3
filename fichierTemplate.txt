#include <iostream>

template<typename T>
class Point {
public:
    T x, y;

    // Constructeur
    Point(T x = T{}, T y = T{}) : x(x), y(y) {}

    // Méthode membre modifiée : version simplifiée sans référence const
    void translate(T dx, T dy) {
        x += dx;
        y += dy;
    }
};

// Opérateur + corrigé : somme (et non soustraction !)
template<typename T>
Point<T> operator+(const Point<T>& a, const Point<T>& b) {
    return Point<T>(a.x + b.x, a.y + b.y); // ✅ correction logique
}

// Opérateur << avec format court (sans "x=" ni "y=")
template<typename T>
std::ostream& operator<<(std::ostream& os, const Point<T>& p) {
    return os << "(" << p.x << ", " << p.y << ")";
}

// Exemple d'utilisation
int main() {
    Point<int> p1(2, 3), p2(4, 5);
    Point<int> p3 = p1 + p2;
    std::cout << p3 << std::endl;  // Affiche : (6, 8)

    Point<double> q1(1.5, 2.5), q2(0.5, 1.0);
    q1.translate(1.0, 1.0);
    std::cout << q1 + q2 << std::endl;  // Affiche : (3.5, 4.5)

    return 0;
}
