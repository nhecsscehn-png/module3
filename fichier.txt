#include <iostream>
#include <vector>

int main() {
    std::vector<int> nombres = {1, 2, 3, 4, 5, 6};
    int somme = 0;

    for (int n : nombres) {
        somme += n;
    }

    std::cout << "Contenu du vecteur : ";
    for (int n : nombres) {
        std::cout << "[" << n << "] ";
    }

    double moyenne = static_cast<double>(somme) / nombres.size();
    std::cout << "\nSomme = " << somme << std::endl;
    std::cout << "Moyenne = " << moyenne << std::endl;

    return 0;
}
